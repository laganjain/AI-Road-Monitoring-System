// esp32_water_sensor.ino
// Demo structure for waterlogging detection using ESP32 + Ultrasonic sensor

#define TRIG_PIN 5
#define ECHO_PIN 18

long duration;
float distanceCm;

void setup() {
  Serial.begin(115200);
  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);
}

void loop() {
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);

  duration = pulseIn(ECHO_PIN, HIGH);
  distanceCm = duration * 0.034 / 2;

  Serial.print("Water depth / distance: ");
  Serial.print(distanceCm);
  Serial.println(" cm");

  // You can set threshold for waterlogging and send payload to backend
  // if(distanceCm < someThreshold) { send alert }

  delay(500);
}
